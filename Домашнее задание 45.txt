

class user {

    constructor (name, surname) {
        this.name = name;
        this.surname = surname;
    }

    getFullName() {
        return this.name + this.surname;
    }
}

class student extends user {
   
    constructor (name, surname, year, this_year) {
        super(name, surname);
        this.year = year;
        this.this_year = this_year;
    }

    getCourse() {
        return this.this_year-this.year
    }
}
